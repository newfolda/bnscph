<script>
    const learnDropdown = document.querySelector('#learn-dropdown ul');
    const countryDropdown = document.querySelector('#country-dropdown [data-dropdown-options]');
    const languageDropdown = document.querySelector('#language-dropdown [data-dropdown-options]');

    function toggleLearn() {
        learnDropdown?.classList.toggle('hidden');

        countryDropdown?.classList.add('hidden');
        countryDropdown?.classList.remove('flex');

        languageDropdown?.classList.add('hidden');
        languageDropdown?.classList.remove('flex');
    }

    function toggleCountry() {
        if (!countryDropdown) return;
        countryDropdown.classList.toggle('hidden');
        countryDropdown.classList.toggle('flex');

        learnDropdown?.classList.add('hidden');
        languageDropdown?.classList.add('hidden');
        languageDropdown?.classList.remove('flex');
    }

    function toggleLanguage() {
        if (!languageDropdown) return;
        languageDropdown.classList.toggle('hidden');
        languageDropdown.classList.toggle('flex');

        learnDropdown?.classList.add('hidden');
        countryDropdown?.classList.add('hidden');
        countryDropdown?.classList.remove('flex');
    }
    document.querySelectorAll('.drawer-side a').forEach((link) => {
        link.addEventListener('click', () => {
            const drawerToggle = document.getElementById('side-nav-drawer');

            if (drawerToggle) {
                drawerToggle.checked = false;
            }

            learnDropdown?.classList.add('hidden');

            countryDropdown?.classList.add('hidden');
            countryDropdown?.classList.remove('flex');

            languageDropdown?.classList.add('hidden');
            languageDropdown?.classList.remove('flex');
        });
    });

    let lastScrollTop = 0;
    const nav = document.getElementById('top-nav');

    window.addEventListener('scroll', () => {
        const currentScroll = window.pageYOffset || document.documentElement.scrollTop;

        // Hide when scrolling down
        if (currentScroll > lastScrollTop && currentScroll > 100) {
            nav.style.transform = 'translateY(-100%)';
        }
        // Show when scrolling up
        else {
            nav.style.transform = 'translateY(0)';
        }

        lastScrollTop = Math.max(currentScroll, 0);
    });
</script>

        // first thing, it will check for first digit

        // length of my - [9, 10]
        // length of ph - [8, 10]
        // length of id - [7, 12]
        function validatePhone(phone, countryCode) {

            var valid = /^\d+$/.test(phone);
            let minLength = true;
            if(countryCode == '+60') {
                var firstDigit = /^1/.test(phone); // Ensure the first digit is 1
                if(phone.length >= 9){
                    minLength = false;
                }
            }else if(countryCode == '+62') {
                var firstDigit = /^8/.test(phone); // Ensure the first digit is 8
                if(phone.length >= 7){
                    minLength = false;
                }
            }else if(countryCode == '+63'){
                var firstDigit = /^9/.test(phone); // Ensure the first digit is 9
                if(phone.length >= 8){
                    minLength = false;
                }
            }else{
                var firstDigit = true;
                minLength = false;
            }
            
            if(!firstDigit){
                isPhoneValid = false;
                if(phone.length > 0){
                    $('#phone').addClass('input-error');
                    $('#phone-error').text("Invalid phone number.").show();
                    $('#sell-submit-form').prop('disabled', true).removeClass('bg-blue-500').addClass('bg-gray-400 cursor-not-allowed');
                }else{
                    $('#phone').addClass('input-error');
                    $('#phone-error').text("Phone number should not be empty.").show();
                    $('#phone-warning').text('').hide();
                    return;
                }
                
            }else{
                if(!valid){
                    isPhoneValid = false;
                    $('#phone').addClass('input-error');
                    $('#phone-error').text("Please enter a valid phone number (digits only).").show();
                    $('#sell-submit-form').prop('disabled', true).removeClass('bg-blue-500').addClass('bg-gray-400 cursor-not-allowed');
                }else{
                    if(minLength){
                        isPhoneValid = false;
                        $('#phone-warning').text("Invalid number (Too short).").show();
                    }else{
                        isPhoneValid = true;
                        $('#phone-warning').text('').hide();
                    }
                    $('#phone').removeClass('input-error');
                    $('#phone-error').text('').hide();
                    $('#sell-submit-form').prop('disabled', false).removeClass('bg-gray-400 cursor-not-allowed').addClass('bg-blue-500');
                }
            }
        }

            function setOptions(selectEl, items, selected) {
            if (!selectEl) return;
            selectEl.innerHTML = '';
            console.log("selected", selected);
            const arr = Array.isArray(items) ? items : [];
            if (arr.length) {
                arr.forEach(it => {
                const value = typeof it === 'object' ? (it.value ?? it.label ?? it) : it;
                const label = typeof it === 'object' ? (it.label ?? it.value ?? it) : it;
                const opt = document.createElement('option');
                if(it != 'Please select'){
                    opt.value = value;
                }
                opt.textContent = label;
                if (selected && (selected === value || selected === label || (selected?.value === value))) {
                    opt.selected = true;
                }
                selectEl.appendChild(opt);
                });
            } else if (selected) { // single selected fallback
                const opt = document.createElement('option');
                opt.value = selected.value ?? selected;
                opt.textContent = selected.label ?? selected;
                opt.selected = true;
                selectEl.appendChild(opt);
            }
            // enable select UI and its icon
            selectEl.removeAttribute('disabled');
            const icon = document.querySelector(`#${selectEl.id}_icon`);
            if (icon) icon.classList.remove('opacity-70');
            }

            // api
            async function runAnalyze(formData, mode = 'photo') {
                const url = (mode === 'grant')
                    ? 'https://ph.mobeecars.com/openai/imgUpload' //grant url
                    : 'https://ph.mobeecars.com/openai/imgUpload'

                console.log("url", url);

                return new Promise((resolve, reject) => {
                    $.ajax({
                    url,
                    method: 'POST',
                    data: formData,
                    processData: false,
                    contentType: false,
                    headers: CSRF_TOKEN ? { 'X-CSRF-TOKEN': CSRF_TOKEN, 'X-Requested-With': 'XMLHttpRequest' } : { 'X-Requested-With': 'XMLHttpRequest' },
                    success(response) {
                        console.log("response", response)
                        resolve(response);
                    },
                    error(jqXHR, textStatus, errorThrown) {  
                        const errInfo = {
                        status: jqXHR.status,
                        textStatus,
                        errorThrown,
                        responseText: jqXHR.responseText,
                        responseJSON: jqXHR.responseJSON
                        };
                        reject(errInfo);
                    },
                    timeout: 30000 // 30s timeout
                    });
                });
            }

<script>
  var acc = document.getElementsByClassName("accordion");

  for (var i = 0; i < acc.length; i++) {
      acc[i].addEventListener("click", function () {
          // Close all other accordions
          for (var j = 0; j < acc.length; j++) {
              if (acc[j] !== this) {
                  acc[j].classList.remove("active");
                  var siblingPanel = acc[j].nextElementSibling;
                  var siblingChevron = acc[j].children[0].children[1];
                  if (siblingPanel && siblingPanel.style.maxHeight) {
                      siblingPanel.style.maxHeight = null;
                      siblingChevron.style.transform = "rotate(0deg)";
                  }
              }
          }

          // Toggle the clicked accordion
          this.classList.toggle("active");
          var panel = this.nextElementSibling;
          var chevron = this.children[0].children[1];
          if (panel.style.maxHeight) {
              panel.style.maxHeight = null;
              chevron.style.transform = "rotate(0deg)";
          } else {
              panel.style.maxHeight = panel.scrollHeight + "px";
              chevron.style.transform = "rotate(180deg)";
          }
      });
  }
